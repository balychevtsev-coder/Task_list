═══════════════════════════════════════════════════════════════════
  📦 ПАКЕТЫ / PACKAGES
═══════════════════════════════════════════════════════════════════

📁 database/
   ├─ __init__.py       - Инициализация пакета
   └─ db.py            - Класс Database для работы с SQLite3
                         • init_db() - создание таблицы
                         • add_task() - добавление задачи
                         • get_all_tasks() - получение всех задач
                         • export_to_csv() - экспорт в CSV
                         • get_task_count() - подсчет задач

📁 handlers/
   ├─ __init__.py       - Экспорт всех роутеров
   ├─ start.py         - Обработчик команды /start
   ├─ add_task.py      - Обработчик команды /add
   └─ list_tasks.py    - Обработчики команд /list и /list_csv

📁 keyboards/
   ├─ __init__.py       - Экспорт функций клавиатур
   └─ main_kb.py       - Основная клавиатура с кнопками
                         • "📋 Список задач"
                         • "➕ Добавить задачу"
                         • "📊 Экспорт CSV"
